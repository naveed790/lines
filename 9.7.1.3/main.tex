\section{Solution}
\textbf{Given:}
\begin{align}
    AD=BC\\
    \angle{CBO}=\angle{DAO}
\end{align}
\textbf{To prove :}
\begin{align}
    \angle{ODA}=\angle{OCB}
\end{align}
\textbf{Proof}\\
The directional vectors are:
\begin{align}
    \vec{m_1}=\Vec{O}-\Vec{D}\\
    \vec{m_2}=\Vec{D}-\Vec{A}
\end{align}
The Normal vectors are:
\begin{align}
    \vec{n_1}=\Vec{O}-\Vec{C}\\
    \vec{n_2}=\Vec{C}-\Vec{B}\\
\end{align}
\begin{align}
    \theta_1 =\cos^-^1\myvec{\frac{m_1^Tm_2}{\norm{m_1}\norm{m_2}}}\\
    \theta_2 =\cos^-^1\myvec{\frac{n_1^Tn_2}{\norm{n_1}\norm{n_2}}}
\end{align}
If $\theta_1 = \theta_2$
\begin{align}
    \triangle{OBC} \cong \triangle{OAD}\\
    OA = OB
\end{align}
